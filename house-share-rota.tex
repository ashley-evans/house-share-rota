% Function: Add Days
\newcount\daycount
\newcommand{\AddDays}[2]{%
	\DTMsavedate{CurrentDate}{#1}%
	\DTMsaveddateoffsettojulianday{CurrentDate}{#2}\daycount%
	\DTMsavejulianday{CurrentDate}{\number\daycount}%
	\DTMusedate{CurrentDate}%
}
